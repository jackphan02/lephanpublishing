/* Example.cpp */
#include "matlab.hpp"
#include "engine.h"

class Test {

public:
void Test::MultipleMatrix() ;

Engine *ep ;

Test () {

if (!(ep = engOpen(NULL))) {
cout<<"Can't start MATLAB engine, check again Project";
exit(-1);

}

}

~Test () {

engClose(ep) ;

}

} ;

/* **************************** */

void Test::MultipleMatrix()
{

/* showing the translations between mwArray and mxArray

procedure : 1.transfer mwArray type into mxArray
then use mxArray to perform the task in Matlab Workspce
2.get results in Matlab workspace then transfer to mxArray
3.transfer mxArray to mwArray type

*/

mwArray mw_A ; /* will define mw_A is a matrix 2 by 4 */
mwArray mw_B ; /* will define mw_B is a matrix 4 by 3 */

int rowA = 2 ;
int colA = 4 ;

int rowB = 4 ;
int colB = 3 ;

mw_A(1,1) = 1 ;
mw_A(1,2) = 2 ;
mw_A(1,3) = 3 ;
mw_A(1,4) = 4 ;

mw_A(2,1) = 5 ;
mw_A(2,2) = 6 ;
mw_A(2,3) = 7 ;
mw_A(2,4) = 8 ;

/* *** */
mw_B(1,1) = 1 ;
mw_B(1,2) = 2 ;
mw_B(1,3) = 3 ;

mw_B(2,1) = 4 ;
mw_B(2,2) = 5 ;
mw_B(2,3) = 6 ;

mw_B(3,1) = 7 ;
mw_B(3,2) = 8 ;
mw_B(3,3) = 9 ;

mw_B(4,1) = 10 ;
mw_B(4,2) = 11 ;
mw_B(4,3) = 12 ;

int rowC = rowA ;
int colC = colB ;


mxArray *mx_A = NULL ;
mxArray *mx_B = NULL ;
mxArray *mx_C = NULL ;


mx_A = mxCreateDoubleMatrix(rowA, colA, mxREAL) ;
mx_B = mxCreateDoubleMatrix(rowB, colB, mxREAL) ;
mx_C = mxCreateDoubleMatrix(rowC, colC, mxREAL) ;


/* begin translation of mwArray to mxArray */

mx_A = mw_A.GetData() ;
mx_B = mw_B.GetData() ;

/* end translation of mwArray to mxArray */


/* begin performing tasks in Matlab workspace
Matlab workspace task : Calculating the product of two matrixes */
/* this is a new function in MATLAB Compiler 3.0 */

engPutVariable(ep, "ml_A", mx_A) ;
engPutVariable(ep, "ml_B", mx_B) ;


engEvalString(ep, " ml_C = mtimes(ml_A,ml_B) ; " ) ;

/* end tasks in Matlab workspace */

/* get results in Matlab workspace then transfer to mxArray */
mx_C = engGetVariable(ep, "ml_C" ) ;


/* begin translation of mxArray to mwArray */

mwArray mw_C (mx_C) ;

/* end translation of mxArray to mwArray */

/*close Matlab workspace */
engEvalString(ep, "close") ;


/* print out */
cout << mw_C << endl ;


}
