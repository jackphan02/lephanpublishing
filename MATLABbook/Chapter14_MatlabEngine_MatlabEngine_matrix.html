void Test::LinearSolve()
{

int i, j ;

double A[3][3] = { {1.1, 5.6, 3.3}, {4.4, 12.3, 6.6} , { 7.7, 8.8, 9.9} } ;
double b[3] = { 12.5, 32.2 , 45.6 } ;

int row = 3 ;
int col = 3 ;

double **db_matrixA ;
double *db_vectorA ;

double **db_matrixU ;
double *db_vectorU ;

double *db_vectorX ;

/* step 1: assign memory for C++ variables */
db_vectorX = new double [row] ;
db_vectorA = new double [row*col] ;
db_vectorU = new double [row*col] ;

db_matrixA = new double*[row] ;
db_matrixU = new double*[row] ;

for(i=0; i<row; i++) {

db_matrixA[i] = new double [col] ;
db_matrixU[i] = new double [col] ;
}

/* assign values for C++ variable */
for(i=0; i<row; i++) {
db_matrixA[i] = & A[i][0] ;
}


/* step 2: declare mxArray variables for using */
mxArray *mx_A = NULL ;
mxArray *mx_b = NULL ;
mxArray *mx_vectorX = NULL ;
mxArray *mx_U = NULL ;


/* step 3: create mxArray vectors and mxArray matrixes*/

mx_A = mxCreateDoubleMatrix(row, col, mxREAL) ;
mx_U = mxCreateDoubleMatrix(row, col, mxREAL) ;
mx_b = mxCreateDoubleMatrix(row, 1 , mxREAL) ;
mx_vectorX = mxCreateDoubleMatrix(row, 1 , mxREAL) ;

/* first, transfer C++ matrix to C++ vector */
DoubleMatrixToDoubleVector (db_matrixA, row, col, db_vectorA) ;


/* second, transfer C++ vector to mxArray matrix */
/* step 4: located memory for mxArray variables
memcpy( (char *)mxGetPr(mx_A), (char *)A, row*col*sizeof(A) ); */

memcpy( (char *)mxGetPr(mx_A), (char *)db_vectorA, row*col*sizeof(double) );
memcpy( (char *)mxGetPr(mx_b), (char *)b, row*sizeof(double) );



/* step 5: transfer name mx_A to ml_A in Matlab workspace
Matlab workspace task : Solve Ax = B, and get vector x and upper matrix U. */

/* step 6: assign variables for object in Matlab workspace */
/* this is a new function in MATLAB Compiler 3.0 (step 5 and step6) */
engPutVariable(ep, "ml_A", mx_A) ;
engPutVariable(ep, "ml_b", mx_b) ;



/* step 7: perform tasks in Matlab Work */
engEvalString(ep, " ml_vectorX = mldivide(ml_A, ml_b) ; " ) ;
engEvalString(ep, " [L, U, P] = lu( ml_A ) ; " ) ;


/* step 8: obtain results form Matlab workspace */
/* this is a new function in MATLAB Compiler 3.0 */
mx_vectorX = engGetVariable(ep, "ml_vectorX" ) ;
mx_U = engGetVariable(ep, "U" ) ;


/* step 9: and tranfer back to mxArray type*/

/* first, transfer a mxArray matrix to C++ vector */
db_vectorX = mxGetPr(mx_vectorX) ;
db_vectorU = mxGetPr(mx_U) ;

/* second, transfer C++ vector to C++ matrix */
DoubleVectorToDoubleMatrix (db_matrixU, row, col, db_vectorU) ;


/* step 10: close Matlab workspace */
engEvalString(ep, "close") ;


/* step 11: free memory */
mxDestroyArray(mx_A) ;
mxDestroyArray(mx_b) ;
mxDestroyArray(mx_vectorX) ;

mxDestroyArray(mx_U) ;


/* step 12: print out */

cout << "\n Solution of equation Ax = b \n\n" ;
for (i=0; i<row; i++) {
cout<< *( db_vectorX + i) << endl ;
}

cout<< "\n Upper Matrix U : \n\n" ;
for (i=0; i<row; i++) {
for (j=0; j<col; j++) {

cout << *( *( db_matrixU +i ) + j ) << "\t\t" ;
}

cout<< endl ;
}


delete [] db_matrixA ;
delete [] db_vectorA ;

delete [] db_matrixU ;
delete [] db_vectorU ;

delete [] db_vectorX ;


}


/* **************************** */

void Test::DoubleMatrixToDoubleVector(double **db_matrix, int row, int col, double *db_vector)
{

int i, j, index ;

for(j=0; j<col; j++) {
for(i=0; i<row; i++) {

index = j*row + i ;
*( db_vector + index ) = *( *( db_matrix + i ) + j ) ;

}
}


}


/* **************************** */

void Test::DoubleVectorToDoubleMatrix(double **db_matrix, int row, int col, double *db_vector)

{
int i, j, index ;

for(j=0; j<col; j++) {
for(i=0; i<row; i++) {

index = j*row + i ;

*( *( db_matrix + i ) + j ) = *( db_vector + index ) ;

}
}


}
